\usepackage{fancyvrb}

\newdimen\MINISIZE
\newcommand{\VRB}[1]{
  \par
  \VerbatimInput[xleftmargin=\fboxsep,xrightmargin=\fboxsep]{#1}
  \par
}

\newcommand{\FRAMEVRB}[1]{
  \par
  \VerbatimInput[xleftmargin=\fboxsep,
                 xrightmargin=\fboxsep,
                 frame=single]{#1}
  \par
}

\newcommand{\FRAMECODE}[1]{
  \MINISIZE=\textwidth\relax
  \advance\MINISIZE-4\fboxsep\relax
  \begin{center}
  \par
  \fbox{%
  \begin{minipage}{\MINISIZE}
  \input{#1}
  \end{minipage}}
  \par
  \end{center}
}

\newcommand{\EXAMPLE}[1]{
  \FRAMEVRB{#1.tex}
  \FRAMECODE{#1.tex}
}

\newcommand{\EXAMPLEA}[1]{
  \MINISIZE=\textwidth\relax
  \advance\MINISIZE-4\fboxsep\relax
  \begin{center}
  \par
  \fbox{%
  \begin{minipage}{\MINISIZE}
  \vskip\fboxsep%
  \VerbatimInput[xleftmargin=\fboxsep,
                 xrightmargin=\fboxsep,
                 frame=single]{#1.tex}
  \input{#1.tex}
  \end{minipage}}
  \par
  \end{center}
}
